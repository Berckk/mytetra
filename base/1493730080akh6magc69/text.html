// Получить структуру ДатаДень = "01", ДатаМесяц = "января", ДатаГод = "2017" из даты документа '01.01.2017'

Функция СтруктураДатыДокумента(ДатаДокумента)

Результат = Новый Структура("ДатаДень, ДатаМесяц, ДатаГод");

ДатаСтрокой = Формат (ДатаДокумента, "ДЛФ=DD");

ДатаСтрокойМассив = СтрРазделить(ДатаСтрокой, " ");

Результат.ДатаДень = ДатаСтрокойМассив[0];

Результат.ДатаМесяц = ДатаСтрокойМассив[1];

Результат.ДатаГод = Сред(ДатаСтрокойМассив[2], 3, 2);

Возврат Результат;

// Получить "Иванов И.И" из "Иванов Иван Иванович"

Функция ФИОСИнициалами(ФИО)

ФИОМассив = СтрРазделить(ФИО, " ");

Если ФИОМассив.Количество() > 2 Тогда

Возврат ФИОМассив[0] + " " + Лев(ФИОМассив[1],1) + "." + Лев(ФИОМассив[2],1) + ".";

ИначеЕсли ФИОМассив.Количество() > 1 Тогда

Возврат ФИОМассив[0] + " " + Лев(ФИОМассив[1],1) + ".";

ИначеЕсли ФИОМассив.Количество() > 0 Тогда

Возврат ФИОМассив[0];

Иначе

Возврат "";

КонецЕсли;
