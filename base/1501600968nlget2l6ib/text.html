Процедура ОбработчикОжидания_ОбновитьИндикаторыВыполненияЭтапов()

    Если НЕ ТипЗнч(СтруктураПараметровВыполнения) = Тип("Структура") Тогда

        Возврат;

    КонецЕсли;    

    

    Если СтруктураПараметровВыполнения.ИндексЭтапа = 1 Тогда

        // Этап 1-2. Формирование аналитик учета, Заполнение документа Отчет производства за смену


        Элементы.НадписьЭтап1.Шрифт = Новый Шрифт(Элементы.НадписьЭтап1.Шрифт, , , Истина);

        Элементы.НадписьЭтап2.Шрифт = Новый Шрифт(Элементы.НадписьЭтап2.Шрифт, , , Истина);

    ИначеЕсли СтруктураПараметровВыполнения.ИндексЭтапа = 3 Тогда        

        // Этап 3. Проведение документа Отчет производства за смену


        Элементы.НадписьЭтап3.Шрифт = Новый Шрифт(Элементы.НадписьЭтап3.Шрифт, , , Истина);

    ИначеЕсли СтруктураПараметровВыполнения.ИндексЭтапа = 4 Тогда        

        // Этап 4. Заполнение и проведение документа Требование-накладная


        Элементы.НадписьЭтап4.Шрифт = Новый Шрифт(Элементы.НадписьЭтап4.Шрифт, , , Истина);

    КонецЕсли;    

        

    ИндикаторЭтапа1 = ?(СтруктураПараметровВыполнения.ИндикаторЭтапа1МаксимальноеЗначение = 0, 0, СтруктураПараметровВыполнения.ИндикаторЭтапа1ТекущееЗначение / СтруктураПараметровВыполнения.ИндикаторЭтапа1МаксимальноеЗначение * 100);

    ИндикаторЭтапа2 = ?(СтруктураПараметровВыполнения.ИндикаторЭтапа2МаксимальноеЗначение = 0, 0, СтруктураПараметровВыполнения.ИндикаторЭтапа2ТекущееЗначение / СтруктураПараметровВыполнения.ИндикаторЭтапа2МаксимальноеЗначение * 100);

    ИндикаторЭтапа3 = СтруктураПараметровВыполнения.ИндикаторЭтапа3;//?(ИндикаторЭтапа3МаксимальноеЗначение = 0, 0, ИндикаторЭтапа3ТекущееЗначение / ИндикаторЭтапа3МаксимальноеЗначение * 100);
